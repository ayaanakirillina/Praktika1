#include <iostream>
using namespace std;
void vvod(float a, float b,float h,float p,float t) {
float P=0;
float V=0; //объем
P=a+b+a+b;
cout<<"Периметр="<<P<<endl;
V=P*h*t; // V-объем стен дома
cout<<"Объем стен дома="<<V<<endl;
float VES=0;
VES=V*p;
cout<<"Вес дома="<<VES<<endl;
}
